import requests
from bs4 import BeautifulSoup
		import pandas as pd
import time
import re

base_url = "https://www.imdb.com/search/title/?groups=top_1000&sort=user_rating,desc&start={}&ref_=adv_nxt"
headers = {'User-Agent': 'Mozilla/5.0'}
all_movies = []

for start in range(1, 301, 50):
    print(f"Scraping page starting from movie {start}...")
    url = base_url.format(start)
    response = requests.get(url, headers=headers)
    soup = BeautifulSoup(response.text, 'html.parser')
    movies = soup.find_all('div', class_='lister-item mode-advanced')

    for movie in movies:
        try:
            year_text = movie.h3.find('span', class_='lister-item-year').text
            year_match = re.search(r'\d{4}', year_text)
            year = int(year_match.group()) if year_match else None
        except:
            year = None

        if year is None or year < 2020 or year > 2025:
            continue

        try:
            title = movie.h3.a.text.strip()
        except:
            title = None

        try:
            genre = movie.p.find('span', class_='genre').text.strip()
        except:
            genre = None

        try:
            runtime = movie.p.find('span', class_='runtime').text.strip().replace("min", "")
        except:
            runtime = None

        try:
            rating = movie.find('div', class_='inline-block ratings-imdb-rating')['data-value']
        except:
            rating = None

        try:
            metascore = movie.find('span', class_='metascore').text.strip()
        except:
            metascore = None

        try:
            desc_tags = movie.find_all('p', class_='text-muted')
            description = desc_tags[1].text.strip()
        except:
            description = None

        try:
            votes = movie.find_all('span', attrs={'name': 'nv'})[0]['data-value'].replace(',', '')
        except:
            votes = None

        try:
            revenue = movie.find_all('span', attrs={'name': 'nv'})[1].text.strip('$M')
        except:
            revenue = None

        try:
            credit = movie.find_all('p')[2].text.strip().split('|')
            if len(credit) == 2:
                director = credit[0].replace('Director:', '').replace('Directors:', '').strip()
                actors = credit[1].replace('Stars:', '').strip()
            else:
                director = credit[0].replace('Director:', '').strip()
                actors = None
        except:
            director, actors = None, None

        all_movies.append({
            'Title': title,
            'Year': year,
            'Genre': genre,
            'Runtime (Minutes)': runtime,
            'Rating': rating,
            'Metascore': metascore,
            'Description': description,
            'Votes': votes,
            'Revenue (Millions)': revenue,
            'Director': director,
            'Actors': actors
        })

        if len(all_movies) >= 300:
            break

    time.sleep(2)

df = pd.DataFrame(all_movies)
df.to_excel("imdb_scrapper.xlsx", index=False)
print("Scraping complete. Data saved to 'imdb_scrapper.xlsx'")
